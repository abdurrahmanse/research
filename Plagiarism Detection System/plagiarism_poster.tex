% Abstract Box
\begin{block}{\Large \textbf{Abstract}}
  This project presents a comprehensive comparison of three plagiarism detection approaches: \textbf{TF-IDF} (traditional ML), \textbf{BiLSTM} (deep learning), and \textbf{BERT} (transformer-based). We evaluate their performance on a curated dataset of 454 text pairs and provide practical insights for deployment scenarios.
  
  \vspace{0.3cm}
  \textbf{Best Performer:} TF-IDF with \textbf{85.51\% accuracy} and \textbf{0.9863 ROC-AUC}
\end{block}

% Dataset Overview Box
\begin{block}{\Large \textbf{Dataset Overview}}
  \begin{itemize}
    \setlength\itemsep{0.8em}
    \item \textbf{Total Samples:} 454 text pairs
    \item \textbf{Plagiarized (Label=1):} 257 pairs (56.6\%)
    \item \textbf{Non-plagiarized (Label=0):} 197 pairs (43.4\%)
    \item \textbf{Train/Val/Test Split:} 70\% / 15\% / 15\%
    \item \textbf{Avg Text Length:} 180 words per sample
  \end{itemize}
  
  \vspace{0.3cm}
  {\small \textit{Balanced dataset with real-world text characteristics}}
\end{block}

% Methodology Box
\begin{block}{\Large \textbf{Methodology}}
  \begin{enumerate}
    \setlength\itemsep{0.6em}
    \item \textbf{Text Preprocessing:} Lowercasing, URL removal, special character handling
    \item \textbf{Feature Engineering:} TF-IDF vectorization for baseline model
    \item \textbf{Model Training:} Three parallel approaches with hyperparameter tuning
    \item \textbf{Evaluation:} Accuracy, Precision, Recall, F1-Score, ROC-AUC
    \item \textbf{Cross-validation:} Stratified k-fold for robust performance estimation
  \end{enumerate}
\end{block}

% Key Findings Box
\begin{alertblock}{\Large \textbf{Key Findings}}
  \begin{itemize}
    \setlength\itemsep{0.8em}
    \item \textbf{TF-IDF wins} with 85.51\% accuracy despite simplicity
    \item \textbf{No one-size-fits-all} solution; choose based on constraints
    \item \textbf{BiLSTM underperforms} on small dataset (454 samples)
    \item \textbf{BERT struggles} with domain-specific fine-tuning
    \item \textbf{Practical insight:} Data size matters more than model complexity
  \end{itemize}
\end{alertblock}

% Recommendation Box
\begin{block}{\Large \textbf{Deployment Recommendations}}
  
  \textbf{\textcolor{accentgreen}{→ Choose TF-IDF if:}}
  \begin{itemize}
    \setlength\itemsep{0.3em}
    \item[$\bullet$] Fast predictions needed
    \item[$\bullet$] Limited resources
    \item[$\bullet$] Model interpretability required
  \end{itemize}
  
  \vspace{0.3cm}
  \textbf{\textcolor{accent}{→ Choose BiLSTM if:}}
  \begin{itemize}
    \setlength\itemsep{0.3em}
    \item[$\bullet$] Balance needed between accuracy \& speed
    \item[$\bullet$] More training data available
    \item[$\bullet$] GPU available
  \end{itemize}
  
  \vspace{0.3cm}
  \textbf{\textcolor{darkblue}{→ Choose BERT if:}}
  \begin{itemize}
    \setlength\itemsep{0.3em}
    \item[$\bullet$] Accuracy is top priority
    \item[$\bullet$] Cloud resources available
    \item[$\bullet$] Batch processing acceptable
  \end{itemize}
\end{block}

% Conclusion Box
\begin{block}{\Large \textbf{Conclusion}}
  This comparative study demonstrates that \textbf{sophisticated models don't guarantee better performance}. TF-IDF's simplicity, efficiency, and high accuracy make it the optimal choice for this dataset. The study highlights the importance of \textbf{empirical evaluation} over assumptions about model superiority.
  
  \vspace{0.3cm}
  \centering \small \textit{\textbf{Data drives decisions, not complexity.}}
\end{block}
