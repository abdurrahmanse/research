\section*{\sectiontitle{Abstract}}

{\small This project compares three plagiarism detection approaches on 454 text pairs. TF-IDF achieves \textbf{85.51\% accuracy}, significantly outperforming BiLSTM (59.42\%) and BERT (56.52\%).}

\vspace{0.15cm}
\fcolorbox{darkblue}{lightblue}{\begin{minipage}{0.98\columnwidth}
\centering\tiny
\textbf{Best Performer:} TF-IDF | \textbf{Accuracy: 85.51\%} | \textbf{ROC-AUC: 98.63\%}
\end{minipage}}

\vspace{0.15cm}
\section*{\sectiontitle{Dataset}}

{\tiny
\begin{itemize}
\item 454 text pairs
\item 257 Plagiarized (56.6\%)
\item 70/15/15 split
\end{itemize}
}

\vspace{0.1cm}
\section*{\sectiontitle{Key Findings}}

{\tiny
\begin{itemize}
\item TF-IDF wins overall
\item Data $>$ complexity
\item No one-size solution
\end{itemize}
}

\section*{\sectiontitle{Models Overview}}

{\tiny
\textbf{\color{accentgreen}TF-IDF}
\begin{itemize}
\item Fast (<100ms)
\item Interpretable
\item Low memory
\end{itemize}

\vspace{0.1cm}
\textbf{\color{accent}BiLSTM}
\begin{itemize}
\item Context aware
\item Slower
\item Better semantics
\end{itemize}

\vspace{0.1cm}
\textbf{\color{darkblue}BERT}
\begin{itemize}
\item Best semantics
\item Pre-trained
\item Slow, High memory
\end{itemize}
}
